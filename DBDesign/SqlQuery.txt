USE SalesDB;

CREATE TABLE VisitTbale
(Visit_id INT IDENTITY(1,1) PRIMARY KEY,
Cust_Name VARCHAR(100),
Contact_Person VARCHAR(100),
Contact_No NUMERIC,
Interest_Product VARCHAR(100),
Visit_Subject VARCHAR(100),
Description VARCHAR(100),
Visit_datetime DATETIME,
Is_Disabled BIT,
Is_Deleted BIT,
Emp_id INT FOREIGN KEY REFERENCES EmployeeRegistration(Emp_id));

CREATE TABLE TBLROLE(
RoleId INT IDENTITY(1,1) PRIMARY KEY,
RoleName VARCHAR(50) NOT NULL);

CREATE TABLE TBLUSER(UserId INT IDENTITY(1,1) PRIMARY KEY,
UserName VARCHAR(20),UserPassword VARCHAR(20),
RoleId INT,
CONSTRAINT FK_LOGIN FOREIGN KEY(RoleId) REFERENCES TBLROLE(RoleId));

CREATE TABLE EMPLOYEEREGISTRATION
(emp_id INT IDENTITY(1,1) PRIMARY KEY,
